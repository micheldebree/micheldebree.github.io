#!/usr/bin/env node
// Save metadata and content for a website as markdown
// ./keep_article.js <url> [tag] [tag] ...

const fs = require('fs');
const jsdom = require('jsdom');
const { exec } = require('child_process');

const { JSDOM } = jsdom;
const { Readability } = require('@mozilla/readability');

const args = process.argv.slice(2);
const url = new URL(args[0]);
const tags = args.slice(1).map((a) => `#${a}`).reduce((r, i) => `${r} ${i}`, '');
const folder = 'Keep/Articles';

function prependFrontmatter(article, markdown) {
  let creator = article.byline ? article.byline : article.site;
  creator = creator || url;
  const tagline = tags.length > 0 ? `Tags: ${tags}` : '';

  return `# ${article.title}

Source: [${creator}](${url})
${tagline}

${markdown}`;
}

function sanitizeFilename(filename) {
  const illegalRe = /[/?<>\\:*|,":!.]/g;
  const controlRe = /[\x00-\x1f\x80-\x9f]/g;
  const reservedRe = /^\.+$/;
  const windowsReservedRe = /^(con|prn|aux|nul|com[0-9]|lpt[0-9])(\..*)?$/i;

  return filename
    .replace(illegalRe, '')
    .replace(controlRe, '')
    .replace(reservedRe, '')
    .replace(windowsReservedRe, '');
}

exec(`wget -k -O tmp.html ${url}`, (wgeterr) => {
  if (wgeterr) throw wgeterr;
  const html = fs.readFileSync('tmp.html').toString();
  const dom = new JSDOM(html);
  const article = new Readability(dom.window.document).parse();

  article.title = article.title.replace(/["]/g, '');
  article.excerpt = article.excerpt.replace(/\n/g, ' ');

  const filenameBase = sanitizeFilename(article.title);

  const htmlFile = `${folder}/${filenameBase}.html`;

  fs.writeFile(htmlFile, article.content, (htmlerr) => {
    if (htmlerr) throw htmlerr;

    const markdownFile = `${folder}/${filenameBase}.md`;

    exec(
      `pandoc -f html -t markdown_strict "${htmlFile}"`,
      (pandocerr, mdstdout) => {
        if (pandocerr) throw pandocerr;

        const markdown = prependFrontmatter(article, mdstdout.toString());
        fs.writeFile(markdownFile, markdown, (markdownerr) => {
          if (markdownerr) throw markdownerr;
        });
      },
    );
  });
});
